7    public class Rectangle { 
8        private Point upperLeft; 
9        private double width; 
10       private double height; 
11    
12       /** 
13        * Create a new rectangle with location and width/height. 
14        * 
15        * @param upperLeft - the upper left pint 
16        * @param width     - widht of rectangle 
17        * @param height    - height of rectangle 
18        */ 
19       public Rectangle(Point upperLeft, double width, double height) { 
20           this.upperLeft = upperLeft; 
21           this.height = height; 
22           this.width = width; 
23       } 
24    
25       /** 
26        * Return a (possibly empty) List of intersection points 
27        * // with the specified line. 
28        * 
29        * @param line - a line 
30        * @return list of intersection points of line with rectangle 
31        */ 
32       public List intersectionPoints(Line line) { 
33           Line up = new Line(this.getUpperRight(), this.getUpperLeft()); 
34           Line down = new Line(this.getLowerRight(), this.getLowerLeft()); 
35           Line left = new Line(this.getUpperLeft(), this.getLowerLeft()); 
36           Line right = new Line(this.getUpperRight(), this.getLowerRight()); 
37           List<Point> intersections = new ArrayList<Point>(); 
38           if (up.isIntersecting(line)) { 
39               intersections.add(up.intersectionWith(line)); 
40           } 
41           if (down.isIntersecting(line)) { 
42               intersections.add(down.intersectionWith(line)); 
43           } 
44           if (left.isIntersecting(line)) { 
45               intersections.add(left.intersectionWith(line)); 
46           } 
47           if (right.isIntersecting(line)) { 
48               intersections.add(right.intersectionWith(line)); 
49           } 
50           return intersections; 
51       } 
52    
53       /** 
54        * Return the width of the rectangle. 
55        * 
56        * @return this.width 
57        */ 
58       public double getWidth() { 
59           return this.width; 
60       } 
61    
62       /** 
63        * Return the height of the rectangle. 
64        * 
65        * @return this.height 
66        */ 
67       public double getHeight() { 
68           return this.height; 
69       } 
70    
71       /** 
72        * Returns the upper-left point of the rectangle. 
73        * 
74        * @return upper left point 
75        */ 
76       public Point getUpperLeft() { 
77           return this.upperLeft; 
78       } 
79    
80       /** 
81        * Returns the upper-right point of the rectangle. 
82        * 
83        * @return upper right point 
84        */ 
85       public Point getUpperRight() { 
86           return new Point(this.getUpperLeft().getX() + this.getWidth(), this.getUpperLeft().getY()); 
87       } 
88    
89       /** 
90        * Returns the lower-left point of the rectangle. 
91        * 
92        * @return lower left point 
93        */ 
94       public Point getLowerLeft() { 
95           return new Point(this.getUpperLeft().getX(), this.getUpperLeft().getY() + this.getHeight()); 
96       } 
97    
98       /** 
99        * Returns the lower right point of the rectangle. 
100       * 
101       * @return lower right point 
102       */ 
103      public Point getLowerRight() { 
104          return new Point(this.getUpperLeft().getX() + this.getWidth(), 
105                  this.getUpperLeft().getY() + this.getHeight()); 
106   
107      }
