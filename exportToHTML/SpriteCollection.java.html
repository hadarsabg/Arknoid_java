3    import java.util.ArrayList; 
4    import java.util.List;

6    /** 
7     * @author Hadar Sabag <hadarsbg@gmail.com>

10    */ 
11   public class SpriteCollection { 
12    
13       private List<Sprite> spritesList = new ArrayList<>(); 
14    
15       /** 
16        * a sprite to the list. 
17        * 
18        * @param s - a sprite object. 
19        */ 
20       public void addSprite(Sprite s) { 
21           this.spritesList.add(s); 
22       } 
23    
24       /** 
25        * remove a given spite from the list. 
26        * 
27        * @param s - a sprite 
28        */ 
29       public void removeSprite(Sprite s) { 
30           this.spritesList.remove(s); 
31       } 
32    
33       /** 
34        * call timePassed() on all sprites. 
35        * @param dt double dt 
36        */ 
37       public void notifyAllTimePassed(double dt) { 
38           // List<Sprite> spritesList1 = new ArrayList<Sprite>(this.spritesList); 
39           for (int i = 0; i < this.spritesList.size(); i++) { 
40               spritesList.get(i).timePassed(dt); 
41           } 
42       } 
43    
44       /** 
45        * call drawOn(d) on all sprites. 
46        * 
47        * @param d - a draw surface 
48        */ 
49       public void drawAllOn(DrawSurface d) { 
50           //List<Sprite> spritesList1 = new ArrayList<Sprite>(this.spritesList); 
51           for (int i = 0; i < this.spritesList.size(); i++) { 
52               spritesList.get(i).drawOn(d); 
53           } 
54       } 
55    
56       /** 
57        * return number of sprites. 
58        * 
59        * @return number of sprites 
60        */ 
61       public int numberOfSprites() { 
62           return this.spritesList.size(); 
63       } 
64   }
