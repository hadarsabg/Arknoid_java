4    public class BallRemover implements HitListener {

8        /** 
9         * constructor.

16           this.remainingBalls = remainingBalls; 
17       } 
18    
19    
20       /** 
21        * remove from game the ball that hit the being hit block. 
22        * also update the game remainingBalls counter. 
23        * 
24        * @param beingHit the block that is being hit 
25        * @param hitter   the Ball that's doing the hitting 
26        */ 
27       public void hitEvent(Block beingHit, Ball hitter) {
