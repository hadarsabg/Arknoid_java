12   public class GameFlow { 
13       private AnimationRunner animationRunner; 
14       private KeyboardSensor keyboardSensor; 
15       private Counter score; 
16       private Counter lives; 
17       private HighScoresTable highScoresTable; 
18    
19    
20       /** 
21        * constructor. 
22        * 
23        * @param ar              animation runner 
24        * @param ks              KeyboardSensor 
25        * @param highScoresTable highscore table 
26        */ 
27       public GameFlow(AnimationRunner ar, KeyboardSensor ks, HighScoresTable highScoresTable) { 
28           this.animationRunner = ar; 
29           this.keyboardSensor = ks; 
30           this.score = new Counter(); 
31           this.lives = new Counter(); 
32           this.highScoresTable = highScoresTable; 
33       } 
34    
35       /** 
36        * run the wanted levels of the game. 
37        * 
38        * @param levels a list of levels to run 
39        */ 
40       public void runLevels(List<LevelInformation> levels) { 
41           /* 
42           HighScoresTable highScoresTable = new HighScoresTable(4); 
43           File file = new File("highscores"); 
44           try { 
45               highScoresTable.load(file); 
46           } catch (IOException e) { 
47           }*/ 
48    
49           EndScreen endScreen = new EndScreen(this.lives, this.score); 
50           KeyPressStoppableAnimation keyPressStoppableEndScreen = 
51                   new KeyPressStoppableAnimation(this.keyboardSensor, "space", endScreen); 
52           int y = this.lives.getValue(); 
53           this.lives.decrease(y); 
54           this.lives.increase(7); 
55           int x = this.score.getValue(); 
56           this.score.decrease(x); 
57           for (LevelInformation levelInfo : levels) { 
58    
59               GameLevel level = new GameLevel( 
60                       levelInfo, this.animationRunner, this.keyboardSensor, this.score, this.lives); 
61               level.initialize(); 
62    
63               //while level has more blocks and player has more lives 
64               while (level.getRemainingBlocks().getValue() != 0 && this.lives.getValue() != 0) { 
65                   level.playOneTurn(); 
66               } 
67    
68               if (this.lives.getValue() == 0) { 
69                   break; 
70               } 
71           } 
72           if (highScoresTable.getRank(this.score.getValue()) < highScoresTable.size()) { 
73               DialogManager dialog = animationRunner.getGui().getDialogManager(); 
74               String name = dialog.showQuestionDialog("Name", "What is your name?", ""); 
75               ScoreInfo scoreInfo = new ScoreInfo(this.score.getValue(), name); 
76               highScoresTable.add(scoreInfo); 
77           } 
78           animationRunner.run(keyPressStoppableEndScreen); 
79           HighScoresAnimation highScoresAnimation = new HighScoresAnimation(highScoresTable); 
80           KeyPressStoppableAnimation keyPressStoppableHighScore 
81                   = new KeyPressStoppableAnimation(this.keyboardSensor, "space", highScoresAnimation); 
82           try { 
83               highScoresTable.save(new File("highscores")); 
84           } catch (IOException e) { 
85               e.printStackTrace(); 
86           } 
87           animationRunner.run(keyPressStoppableHighScore); 
88       } 
89   }
