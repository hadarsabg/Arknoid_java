30       /** 
31        * Assume an object moving from line.start() to line.end(). 
32        * If this object will not collide with any of the collidables 
33        * in this collection, return null. Else, return the information 
34        * about the closest collision that is going to occur. 
35        * 
36        * @param trajectory - the line of the ball movement 
37        * @return closest collision info 
38        */ 
39       public CollisionInfo getClosestCollision(Line trajectory) { 
40           //if there are no obstacle in the environment 
41           if (obstacles.isEmpty()) { 
42               return null; 
43           } 
44           double minDistance; 
45           Point p, p2; 
46           Collidable c, c2; 
47           p = trajectory.closestIntersectionToStartOfLine(obstacles.get(0).getCollisionRectangle()); 
48           c = obstacles.get(0); 
49           //if there are no collision at all 
50           for (int i = 1; i < obstacles.size(); i++) { 
51               p2 = trajectory.closestIntersectionToStartOfLine(obstacles.get(i).getCollisionRectangle()); 
52               c2 = obstacles.get(i); 
53               if (p == null) { 
54                   p = p2; 
55                   c = c2; 
56               } else if (p2 != null) { 
57                   if (p2.distance(trajectory.start()) < p.distance(trajectory.start())) { 
58                       p = p2; 
59                       c = c2; 
60                   } 
61               } 
62           } 
63           if (p == null) { 
64               return null; 
65           } 
66           return new CollisionInfo(p, c); 
67       } 
68   }
