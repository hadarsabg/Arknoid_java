155      /** 
156       * change the ball position. 
157       * 
158       *@param dt  -double dt 
159       */ 
160      public void moveOneStep(double dt) { 
161          Velocity v1 = new Velocity(this.getVelocity().getDx() * dt, this.getVelocity().getDy() * dt); 
162          Line trajectory = new Line(this.center, v1.applyToPoint(this.center)); 
163          // Line trajectory = new Line(this.center, 
164          //  new Point(this.center.getX() + this.getVelocity().getDx(), 
165          //        this.center.getY() + this.getVelocity().getDy())); 
166          CollisionInfo collision = this.getEnvironment().getClosestCollision(trajectory); 
167          if (collision == null) { 
168              this.center = v1.applyToPoint(this.center); 
169              return; 
170          } 
171          Rectangle rectangle = collision.collisionObject().getCollisionRectangle(); 
172   
173   
174          //if hits corners of block 
175   
176          if (collision.collisionPoint().equals(rectangle.getUpperRight()) 
177                  || collision.collisionPoint().equals(rectangle.getUpperLeft()) 
178                  || collision.collisionPoint().equals(rectangle.getLowerRight()) 
179                  || rectangle.getUpperLeft().equals(collision.collisionPoint())) { 
180              Velocity v = collision.collisionObject().hit(this, collision.collisionPoint(), this.getVelocity()); 
181              this.setVelocity(v); 
182              return; 
183          } 
184          // if hits upper line 
185          if (collision.collisionPoint().getY() == rectangle.getUpperLeft().getY()) { 
186              this.center = new Point(collision.collisionPoint().getX(), 
187                      collision.collisionPoint().getY() - this.getSize()); 
188              Velocity v = collision.collisionObject().hit(this, collision.collisionPoint(), this.getVelocity()); 
189              this.setVelocity(v); 
190              return; 
191   
192          } 
193          //if hits lower line 
194          if (collision.collisionPoint().getY() == rectangle.getLowerRight().getY()) { 
195              this.center = new Point(collision.collisionPoint().getX(), 
196                      collision.collisionPoint().getY() + this.getSize()); 
197              Velocity v = collision.collisionObject().hit(this, collision.collisionPoint(), this.getVelocity()); 
198              this.setVelocity(v); 
199              return; 
200   
201          } 
202          //if hits right side 
203          if (collision.collisionPoint().getX() == rectangle.getUpperRight().getX()) { 
204              this.center = new Point(collision.collisionPoint().getX() + this.getSize(), 
205                      collision.collisionPoint().getY()); 
206              Velocity v = collision.collisionObject().hit(this, collision.collisionPoint(), this.getVelocity()); 
207              this.setVelocity(v); 
208              return; 
209   
210          } 
211          //if hits left side 
212          if (collision.collisionPoint().getX() == rectangle.getUpperLeft().getX()) { 
213              this.center = new Point(collision.collisionPoint().getX() - this.getSize() 
214                      , collision.collisionPoint().getY()); 
215              ; 
216              Velocity v = collision.collisionObject().hit(this, collision.collisionPoint(), this.getVelocity()); 
217              this.setVelocity(v); 
218              return; 
219   
220          } 
221   
222      }

224      /** 
225       * return the ball speed. 
226       * 
227       * @return 100/ ball radius - if radius is less then 50, else 2 
228       */ 
229      public double getBallSpeed() { 
230          if (this.getSize() >= 50) { 
231              return 2; 
232          } 
233          return 100 / this.getSize(); 
234      }
