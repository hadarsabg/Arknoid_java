19    
20       /** 
21        * end the program. 
22        */ 
23       public void exit() { 
24           System.exit(-1); 
25       } 
26    
27       /** 
28        * main method-starts a new game. 
29        * run the wanted levels of the game, 
30        * if no levels are entered- run all four levels. 
31        * 
32        * @param args - list of numbers of levels to play 
33        */ 
34       public static void main(String[] args) { 
35           Ass6Game ass6Game = new Ass6Game(); 
36           //create highScore table 
37           File file = new File("highscores"); 
38           HighScoresTable highScoresTable = new HighScoresTable(4); 
39    
40           try { 
41               highScoresTable.load(file); 
42           } catch (IOException e) { 
43               ; 
44           } 
45           AnimationRunner animationRunner = new AnimationRunner(); 
46           KeyboardSensor keyboardSensor = animationRunner.getGui().getKeyboardSensor(); 
47           GameFlow gameFlow = new GameFlow(animationRunner, keyboardSensor, highScoresTable); 
48           InputStream inputStream1; 
49           if (args.length != 0) { 
50               inputStream1 = ClassLoader.getSystemClassLoader().getResourceAsStream(args[0]); 
51           } else { 
52               inputStream1 = ClassLoader.getSystemClassLoader().getResourceAsStream("level_sets.txt"); 
53           } 
54           Reader reader = new InputStreamReader(inputStream1); 
55           LevelsSetsReader levelsSetsReader = new LevelsSetsReader(); 
56           Map<String, String> levelsSet = levelsSetsReader.levelSetsFromReader(reader); 
57           Menu<Task<Void>> subMenu = new MenuAnimation<Task<Void>>("choose level", keyboardSensor); 
58           Menu<Task<Void>> menu = new MenuAnimation<Task<Void>>("arkanoid", keyboardSensor); 
59           for (String key : levelsSet.keySet()) { 
60               String[] nameSeparate = key.split(":"); 
61               String symbol = nameSeparate[0]; 
62               String setsName = nameSeparate[1]; 
63               String filePath = levelsSet.get(key); 
64               InputStream inputStream = ClassLoader.getSystemClassLoader().getResourceAsStream(filePath); 
65               Reader r = new InputStreamReader(inputStream); 
66    
67               LevelSpecificationReader levelSpecificationReader = new LevelSpecificationReader(); 
68               List<LevelInformation> levelsListOfSet = new ArrayList<>(); 
69               levelsListOfSet = levelSpecificationReader.fromReader(r); 
70               List<LevelInformation> finalLevelsListOfSet = levelsListOfSet; 
71    
72               Task<Void> runLevelsTask = new Task<Void>() { 
73                   @Override 
74                   public Void run() { 
75    
76                       gameFlow.runLevels(finalLevelsListOfSet); 
77                       return null; 
78                   } 
79               }; 
80               subMenu.addSelection(symbol, 
81                       "press \"" + symbol + "\" for " + setsName + " level", runLevelsTask); 
82           } 
83    
84           Task<Void> subMenuTask = new Task<Void>() { 
85               @Override 
86               public Void run() { 
87                   animationRunner.run(subMenu); 
88                   Task<Void> task = subMenu.getStatus(); 
89                   task.run(); 
90                   return null; 
91               } 
92           }; 
93           menu.addSelection("s", "Press \"s\" to start a new game", subMenuTask); 
94    
95           HighScoresAnimation highScoresAnimation = new HighScoresAnimation(highScoresTable); 
96           KeyPressStoppableAnimation keyPressStoppableHighScore 
97                   = new KeyPressStoppableAnimation(keyboardSensor, "space", highScoresAnimation); 
98    
99           Task<Void> highScoresTask = new Task<Void>() { 
100              @Override 
101              public Void run() { 
102                  animationRunner.run(keyPressStoppableHighScore); 
103                  return null; 
104              } 
105          }; 
106   
107          Task<Void> quitTask = new Task<Void>() { 
108              @Override 
109              public Void run() { 
110                  ass6Game.exit(); 
111                  return null; 
112              } 
113          }; 
114   
115          menu.addSelection("h", "Press \"h\" to see the high scores", highScoresTask); 
116          menu.addSelection("q", "Press \"q\" to quit", quitTask); 
117   
118          while (true) { 
119              animationRunner.run(menu); 
120              Task<Void> task = menu.getStatus(); 
121              task.run(); 
122          } 
123      } 
124   
125  }
