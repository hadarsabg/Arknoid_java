12   public class Paddle implements Sprite, Collidable { 
13       private biuoop.KeyboardSensor keyboard; 
14       private Rectangle paddleRect; 
15       private GameLevel game; 
16       private int rightBorder; 
17       private int leftBorder; 
18       private Color color; 
19       private int speed; 
20    
21    
22       /** 
23        * class constructor. 
24        * 
25        * @param rectangle   - the paddle shape 
26        * @param game        - a game object 
27        * @param rightBorder - the game surface left border 
28        * @param leftBorder  - the game surface right border 
29        * @param runner      - animation runner 
30        * @param color       - the color of the paddle 
31        * @param speed       - the paddle speed 
32        */ 
33       public Paddle(Rectangle rectangle, GameLevel game, int rightBorder, int leftBorder, 
34                     AnimationRunner runner, Color color, int speed) { 
35           this.paddleRect = rectangle; 
36           this.game = game; 
37           this.keyboard = runner.getGui().getKeyboardSensor(); 
38           this.rightBorder = rightBorder; 
39           this.leftBorder = leftBorder; 
40           this.color = color; 
41           this.speed = speed; 
42       } 
43    
44       /** 
45        * if paddle dosent exit left border move its location to the left. 
46        * @param dt double dt 
47        */ 
48       public void moveLeft(double dt) { 
49           if (this.paddleRect.getUpperLeft().getX() >= this.leftBorder) { 
50               Point newLocation = 
51                       new Point(this.paddleRect.getUpperLeft().getX() - speed * dt, 
52                               this.paddleRect.getUpperLeft().getY()); 
53               this.paddleRect = new Rectangle(newLocation, this.paddleRect.getWidth(), this.paddleRect.getHeight()); 
54           } 
55       } 
56    
57       /** 
58        * if paddle dosent exit right border move its location to the right. 
59        * @param dt double dt 
60        */ 
61       public void moveRight(double dt) { 
62           if (this.paddleRect.getUpperRight().getX() <= this.rightBorder) { 
63               Point newLocation = 
64                       new Point(this.paddleRect.getUpperLeft().getX() + speed * dt, 
65                               this.paddleRect.getUpperLeft().getY()); 
66               this.paddleRect = new Rectangle(newLocation, this.paddleRect.getWidth(), this.paddleRect.getHeight()); 
67           } 
68       } 
69    
70       /** 
71        * sprite method - move left or right according to keyboard arrows. 
72        * @param dt double dt 
73        */ 
74       public void timePassed(double dt) { 
75           if (this.keyboard.isPressed(KeyboardSensor.LEFT_KEY)) { 
76               this.moveLeft(dt); 
77           } else if (this.keyboard.isPressed(KeyboardSensor.RIGHT_KEY)) { 
78               this.moveRight(dt); 
79           } 
80       } 
81    
82       /** 
83        * draw the paddle on a given surface. 
84        * 
85        * @param d - adraw surface 
86        */ 
87       public void drawOn(DrawSurface d) { 
88           d.setColor(this.color); 
89           d.fillRectangle((int) this.paddleRect.getUpperLeft().getX(), (int) this.paddleRect.getUpperLeft().getY(), 
90                   (int) this.paddleRect.getWidth(), (int) this.paddleRect.getHeight()); 
91       }

104      /** 
105       * return the new ball velocity after the ball hit paddle. 
106       * 
107       * @param collisionPoint  - collision point with the collidable shape 
108       * @param currentVelocity - the ball velocity 
109       * @param hitter          - the ball that is doing the hit 
110       * @return the new velocity 
111       */ 
112      public Velocity hit(Ball hitter, Point collisionPoint, Velocity currentVelocity) { 
113          double regionWidth = this.paddleRect.getWidth() / 5; 
114          double upperLeftX = this.paddleRect.getUpperLeft().getX(); 
115          //if hits upper side 
116          if (collisionPoint.getY() == paddleRect.getUpperLeft().getY()) { 
117              //if hits 1st region 
118              if (collisionPoint.getX() >= upperLeftX && collisionPoint.getX() <= upperLeftX + regionWidth) { 
119                  Velocity v = Velocity.fromAngleAndSpeed(300, currentVelocity.getSpeed()); 
120                  return v; 
121                  //if hits 2nd region 
122              } else if (collisionPoint.getX() >= upperLeftX + regionWidth 
123                      && collisionPoint.getX() <= upperLeftX + (2 * regionWidth)) { 
124                  Velocity v = Velocity.fromAngleAndSpeed(330, currentVelocity.getSpeed()); 
125                  return v; 
126                  //if hits 3 region 
127              } else if (collisionPoint.getX() >= upperLeftX + (2 * regionWidth) 
128                      && collisionPoint.getX() <= upperLeftX + (3 * regionWidth)) { 
129                  Velocity v = new Velocity(currentVelocity.getDx(), (-1) * currentVelocity.getDy()); 
130                  return v; 
131                  //4 region 
132              } else if (collisionPoint.getX() >= upperLeftX + (3 * regionWidth) 
133                      && collisionPoint.getX() <= upperLeftX + (4 * regionWidth)) { 
134                  Velocity v = Velocity.fromAngleAndSpeed(30, currentVelocity.getSpeed()); 
135                  return v; 
136                  //5 region 
137              } else if (collisionPoint.getX() >= upperLeftX + (4 * regionWidth) 
138                      && collisionPoint.getX() <= upperLeftX + (5 * regionWidth)) { 
139                  Velocity v = Velocity.fromAngleAndSpeed(60, currentVelocity.getSpeed()); 
140                  return v; 
141              } 
142          } 
143          //if hits lower side 
144          if (collisionPoint.getY() == paddleRect.getLowerLeft().getY()) { 
145              return new Velocity(currentVelocity.getDx(), (-1) * currentVelocity.getDy()); 
146          } 
147          //if hits left or right side 
148          if (collisionPoint.getX() == paddleRect.getUpperLeft().getX() 
149                  || collisionPoint.getX() == paddleRect.getUpperRight().getX()) { 
150              return new Velocity((-1) * currentVelocity.getDx(), currentVelocity.getDy()); 
151          } 
152          //if doesn't hit paddle 
153          return currentVelocity; 
154      }

158       * 
159       * @param g - a game object 
160       */ 
161      public void addToGame(GameLevel g) { 
162          g.addSprit(this); 
163          g.addCollidable(this); 
164      }

166      /** 
167       * remove this paddle from  game. 
168       * 
169       * @param g - a game object 
170       */ 
171      public void removeFromGame(GameLevel g) { 
172          g.removeSprite(this); 
173          g.removeCollidable(this); 
174      }

176      /** 
177       * move paddle to center point. 
178       * 
179       * @param center - the center point of the surface 
180       */ 
181      public void moveToCenter(Point center) { 
182          this.paddleRect = new Rectangle(center, this.paddleRect.getWidth(), this.paddleRect.getHeight()); 
183      }
