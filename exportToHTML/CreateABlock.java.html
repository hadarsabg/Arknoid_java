1    import java.awt.Color; 
2    import java.awt.Image; 
3    import java.util.Map; 
4    import java.util.TreeMap;

9    public class CreateABlock implements BlockCreator { 
10       private int width; 
11       private int height; 
12       private Map<Integer, Color> hitPointsToColors = new TreeMap<>(); 
13       private Map<Integer, Image> hitPointsToImages = new TreeMap<>(); 
14       private int hitPoints; 
15       private Color stroke = null; 
16    
17       /** 
18        * constructor. 
19        * 
20        * @param width             block width 
21        * @param height            block height 
22        * @param hitPointsToColors maps hit points to colors of block 
23        * @param hitPointsToImages maps hit points to image fill of block 
24        * @param hitPoints         hit points 
25        * @param stroke            color of block stroke 
26        */ 
27       public CreateABlock(int width, int height, Map<Integer, Color> hitPointsToColors, 
28                           Map<Integer, Image> hitPointsToImages, int hitPoints, Color stroke) { 
29           this.height = height; 
30           this.width = width; 
31           this.hitPoints = hitPoints; 
32           this.hitPointsToColors = hitPointsToColors; 
33           this.hitPointsToImages = hitPointsToImages; 
34           this.stroke = stroke; 
35       } 
36    
37       @Override 
38       public Block create(int xpos, int ypos) { 
39           Point pos = new Point(xpos, ypos); 
40           Rectangle blockRectangle = new Rectangle(pos, this.width, this.height); 
41           Block b = new Block(blockRectangle, this.hitPoints, this.hitPointsToColors, 
42                   this.hitPointsToImages, this.stroke); 
43           return b; 
44           // return new Block(blockRectangle, this.hitPoints, this.hitPointsToColors, this.hitPointsToImages); 
45       } 
46   }
