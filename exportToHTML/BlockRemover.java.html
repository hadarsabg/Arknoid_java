5    // a BlockRemover is in charge of removing blocks from the game, as well as keeping count 
6    // of the number of blocks that remain. 
7    public class BlockRemover implements HitListener { 
8        private GameLevel game; 
9        private Counter remainingBlocks; 
10    
11       /** 
12        * constructor. 
13        * 
14        * @param game          - a game 
15        * @param removedBlocks - the blocks counter from the game 
16        */ 
17       public BlockRemover(GameLevel game, Counter removedBlocks) { 
18           this.game = game; 
19           this.remainingBlocks = removedBlocks; 
20       } 
21    
22       /** 
23        * remove from game the blocks that are hit and reach 0 hit-points. 
24        * also update the remainingBlocks counter, 
25        * and remove this hit listener from the being hit block. 
26        * 
27        * @param beingHit the block that is being hit 
28        * @param hitter   the Ball that's doing the hitting 
29        */ 
30       public void hitEvent(Block beingHit, Ball hitter) { 
31           if (beingHit.getHitPoints() == 0) { 
32               beingHit.removeFromGame(this.game); 
33               beingHit.removeHitListener(this); 
34               this.remainingBlocks.decrease(1); 
35           } 
36       } 
37   }
