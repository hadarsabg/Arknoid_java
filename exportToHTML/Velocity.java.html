7    // Velocity specifies the change in position on the `x` and the `y` axes. 
8    public class Velocity { 
9        private double dx; 
10       private double dy; 
11    
12       /** 
13        * class constructor. 
14        * 
15        * @param dx - change on x axis 
16        * @param dy - change on y axis 
17        */ 
18       public Velocity(double dx, double dy) { 
19           this.dx = dx; 
20           this.dy = dy; 
21       } 
22    
23       /** 
24        * return change on x axis. 
25        * 
26        * @return this.dx -change on x axis 
27        */ 
28       public double getDx() { 
29           return this.dx; 
30       } 
31    
32       /** 
33        * return change on y axis. 
34        * 
35        * @return this.dy -change on y axis 
36        */ 
37       public double getDy() { 
38           return this.dy; 
39    
40       } 
41    
42       /** 
43        * update a point coordinates. 
44        * Take a point with position (x,y) and return a new point 
45        * with position (x+dx, y+dy) 
46        * 
47        * @param p - point 
48        * @return point (x+dx, y+dy) 
49        */ 
50       public Point applyToPoint(Point p) { 
51           return new Point(p.getX() + this.getDx(), p.getY() + this.getDy()); 
52       } 
53    
54       /** 
55        * Converting from polar to Cartesian coordinates. 
56        * 
57        * @param angle - angle of move 
58        * @param speed - double 
59        * @return v -velocity 
60        */ 
61       public static Velocity fromAngleAndSpeed(double angle, double speed) { 
62           double dx = speed * Math.sin(Math.toRadians(angle)); 
63           double dy = (-1) * speed * Math.cos(Math.toRadians(angle)); 
64           return new Velocity(dx, dy); 
65       } 
66    
67       /** 
68        * return the angle of the velocity. 
69        * @return angle of velocity. 
70        */ 
71       public double getAngle() { 
72           return this.dy / this.dx; 
73       } 
74    
75       /** 
76        * return the speed vector of the velocity. 
77        * @return speed vector. 
78        */ 
79       public double getSpeed() { 
80           return Math.sqrt((this.dx * this.dx) + (this.dy * this.dy)); 
81       } 
82    
83    
84   }
