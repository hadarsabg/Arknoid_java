80       /** 
81        * Initialize a new game: create the Blocks and Ball (and Paddle) 
82        * and add them to the game. 
83        */ 
84       public void initialize() { 
85           //set the background 
86           Sprite levelName = new Sprite() { 
87               @Override 
88               public void drawOn(DrawSurface d) { 
89                   d.setColor(new Color(0xFCEAFF)); 
90                   d.fillRectangle(0, 0, d.getWidth(), 22); 
91                   d.setColor(Color.black); 
92                   d.drawText(590, 15, levelInformation.levelName(), 20); 
93               } 
94               @Override 
95               public void timePassed(double dt) { 
96                   return; 
97               } 
98           }; 
99    
100          this.addSprit(this.levelInformation.getBackground()); 
101          this.addSprit(levelName); 
102          ScoreIndicator scoreIndicator = new ScoreIndicator(this.score); 
103          LivesIndicator livesIndicator = new LivesIndicator(this.lives); 
104          this.addSprit(scoreIndicator); 
105          this.addSprit(livesIndicator); 
106          this.paddle.addToGame(this); 
107          int surfaceWidth = this.runner.getGui().getDrawSurface().getWidth(); 
108          int surfaceHeight = this.runner.getGui().getDrawSurface().getHeight(); 
109          //set the blocks of the borders 
110          Block up = new Block(new Rectangle(new Point(0, 20), surfaceWidth, 30), 
111                  0, Color.gray); 
112          Block left = new Block(new Rectangle(new Point(0, 20), 30, surfaceHeight), 
113                  0, Color.gray); 
114          Block right = new Block(new Rectangle(new Point(surfaceWidth - 30, 20), 
115                  30, surfaceHeight), 0, Color.gray); 
116          up.addToGame(this); 
117          left.addToGame(this); 
118          right.addToGame(this); 
119          //set death region in the bottom 
120          BallRemover ballRemover = new BallRemover(this, this.remainingBalls); 
121          Block deathRegion = new Block(new Rectangle(new Point(0, surfaceHeight), surfaceWidth, 30), 
122                  0, Color.gray); 
123          deathRegion.addHitListener(ballRemover); 
124          deathRegion.addToGame(this); 
125          //set the small blocks in the middle 
126          BlockRemover blockRemover = new BlockRemover(this, this.remainingBlocks); 
127          ScoreTrackingListener scoreTrackingListener = new ScoreTrackingListener(this.score); 
128          for (int i = 0; i < this.levelInformation.numberOfBlocksToRemove(); i++) { 
129              Block block = this.levelInformation.blocks().get(i); 
130              block.addHitListener(blockRemover); 
131              block.addHitListener(scoreTrackingListener); 
132              block.addToGame(this); 
133              this.remainingBlocks.increase(1); 
134          } 
135      }

146      /** 
147       * move the paddle to the center and create the balls. 
148       */ 
149      public void createBallsOnTopOfPaddle() { 
150          this.paddle.moveToCenter(new Point((int) (400 - (this.levelInformation.paddleWidth() / 2)), 540)); 
151          for (int i = 0; i < this.levelInformation.numberOfBalls(); i++) { 
152              Ball ball = new Ball(400, 535, 6, Color.WHITE, this.environment); 
153              ball.addToGame(this); 
154              ball.setVelocity(levelInformation.initialBallVelocities().get(i)); 
155              this.remainingBalls.increase(1); 
156          } 
157      }

159      /** 
160       * Run one turn of the game. 
161       * start the animation loop using the runner. 
162       */ 
163      public void playOneTurn() { 
164          this.createBallsOnTopOfPaddle(); 
165          this.runner.run(new CountdownAnimation(2, 3, this.sprites)); 
166          this.running = true; 
167          this.runner.run(this); 
168      } 
169   
170      /** 
171       * the animation logic. 
172       * draw the sprites and notify time passed, 
173       * check the counters, and change running(stop condition) 
174       * according to them. 
175       * also check if game needs to be paused. 
176       * 
177       * @param d  draw surface 
178       * @param dt -double dt 
179       */ 
180      public void doOneFrame(DrawSurface d, double dt) { 
181          this.sprites.drawAllOn(d); 
182          this.sprites.notifyAllTimePassed(dt); 
183          if (this.remainingBlocks.getValue() == 0) { 
184              this.score.increase(100); 
185              this.running = false; 
186          } 
187          if (this.remainingBalls.getValue() == 0) { 
188              this.lives.decrease(1); 
189              this.running = false; 
190          } 
191   
192          if (this.keyboardSensor.isPressed("p")) { 
193              this.runner.run(new KeyPressStoppableAnimation(this.keyboardSensor, 
194                      "space", new PauseScreen())); 
195          } 
196      }
