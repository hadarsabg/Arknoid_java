39       @Override 
40       public List<Velocity> initialBallVelocities() { 
41           List<Velocity> list = new ArrayList<Velocity>(); 
42           String[] velocities = map.get("ball_velocities").split(" "); 
43           for (int i = 0; i < velocities.length; i++) { 
44               String[] v = velocities[i].split(","); 
45               Velocity velocity = Velocity.fromAngleAndSpeed(Integer.parseInt(v[0]), Integer.parseInt(v[1])); 
46               list.add(velocity); 
47           } 
48           return list; 
49       }

67       @Override 
68       public Sprite getBackground() { 
69           Sprite background = null; 
70           String backgroundString = map.get("background"); 
71           //if background is an image- 
72           if (backgroundString.startsWith("image")) { 
73               String imagePath = backgroundString.substring(6, backgroundString.length() - 1); 
74               System.out.println(imagePath); 
75               Image img = null; 
76               try { 
77                   InputStream inputStream = ClassLoader.getSystemClassLoader().getResourceAsStream(imagePath); 
78                   img = ImageIO.read(inputStream); 
79               } catch (IOException e) { 
80                   e.printStackTrace(); 
81               } 
82               Image finalImg = img; 
83               background = new Sprite() { 
84                   @Override 
85                   public void drawOn(DrawSurface d) { 
86                       // Draw the image on a DrawSurface 
87                       d.drawImage(0, 0, finalImg); // draw the image at location 10, 20. 
88                   } 
89    
90                   @Override 
91                   public void timePassed(double dt) { 
92                       return; 
93                   } 
94               }; 
95               return background; 
96           } 
97    
98           // if the background is a color- 
99           if (backgroundString.startsWith("color")) { 
100              String colorString = backgroundString.substring(6, backgroundString.length() - 1); 
101              // set the color and create a sprite of background 
102              ColorsParser colorsParser = new ColorsParser(); 
103              Color color = colorsParser.colorFromString(colorString); 
104              background = new Sprite() { 
105                  @Override 
106                  public void drawOn(DrawSurface d) { 
107                      d.setColor(color); 
108                      d.fillRectangle(0, 0, d.getWidth(), d.getHeight()); 
109                  } 
110   
111                  @Override 
112                  public void timePassed(double dt) { 
113                      return; 
114                  } 
115              }; 
116          } 
117          return background;

121      @Override 
122      public List<Block> blocks() { 
123   
124          List<Block> blockList = new ArrayList<Block>(); 
125          BlocksDefinitionReader blocksDefinitionReader = new BlocksDefinitionReader(); 
126          InputStream is = ClassLoader.getSystemClassLoader().getResourceAsStream(map.get("block_definitions")); 
127          Reader r = new InputStreamReader(is); 
128          BlocksFromSymbolsFactory factory = 
129                  blocksDefinitionReader.fromReader(r); 
130          int xpos = Integer.parseInt(map.get("blocks_start_x")); 
131          int ypos = Integer.parseInt(map.get("blocks_start_y")); 
132          int rowHeight = Integer.parseInt(map.get("row_height")); 
133          for (int i = 0; i < this.blocklayout.size(); i++) { 
134              int x = xpos; 
135              for (int j = 0; j < this.blocklayout.get(i).length(); j++) { 
136                  String currentSymbol = String.valueOf(blocklayout.get(i).charAt(j)); 
137                  if (factory.isSpaceSymbol(currentSymbol)) { 
138                      x = x + factory.getSpaceWidth(currentSymbol); 
139                  } else { 
140                      if (factory.isBlockSymbol(currentSymbol)) { 
141                          Block block = factory.getBlock(currentSymbol, x, ypos); 
142                          blockList.add(block); 
143                          x = (int) (x + block.getCollisionRectangle().getWidth()); 
144                      } 
145                  } 
146              } 
147              ypos = ypos + rowHeight; 
148   
149          } 
150          return blockList; 
151      }
