38       @Override 
39       public void doOneFrame(DrawSurface d, double dt) { 
40           d.setColor(Color.black); 
41           d.fillRectangle(0, 0, 800, 600); 
42           int blockWidth = 50; 
43           int blockHeight = 35; 
44           java.util.List<Color> colors = new ArrayList<>(); 
45           colors.add(Color.cyan); 
46           colors.add(Color.yellow); 
47           colors.add(Color.orange); 
48           colors.add(Color.green); 
49           colors.add(Color.MAGENTA); 
50    
51           int hits = 2; 
52           int row = 7; 
53           for (int j = 1; j <= 5; j++) { 
54               Color c = colors.get(j - 1); 
55               for (int i = 1; i < row; i++) { 
56                   if (!((i == 4 && j == 1) || (j == 4 && i == 8) || (j == 3 && i == 8))) { 
57                       d.setColor(c); 
58                       d.fillRectangle(800 - (i * blockWidth), 390 + (j * blockHeight), blockWidth, blockHeight); 
59                       d.setColor(Color.black); 
60                       d.drawRectangle(800 - (i * blockWidth), 390 + (j * blockHeight), blockWidth, blockHeight); 
61                   } 
62               } 
63               row++; 
64           } 
65    
66           d.setColor(Color.magenta); 
67           d.drawText(300, 100, this.menuTitle, 55); 
68           d.setColor(new Color(0xF8CFB4)); 
69           d.fillCircle(600, 200, 9); 
70           d.fillCircle(100, 100, 9); 
71           for (int i = 0; i < this.messages.size(); i++) { 
72               d.drawText(100, 250 + (i * 50), this.messages.get(i), 25); 
73           } 
74           Object[] keys = this.optionalTasks.keySet().toArray(); 
75           for (int j = 0; j < keys.length; j++) { 
76               if (keyboardSensor.isPressed((String) keys[j])) { 
77                   this.status = optionalTasks.get(keys[j]); 
78                   this.stop = true; 
79               } 
80           } 
81       }

83       @Override 
84       public boolean shouldStop() { 
85           if (this.stop) { 
86               this.stop = false; 
87               return true; 
88           } 
89           return this.stop; 
90       }

93       public void addSelection(String key, String message, Object returnVal) {

95           this.messages.add(message); 
96       }
