7    public class Point { 
8        private double x; 
9        private double y; 
10    
11       /** 
12        * class constructor. 
13        * 
14        * @param x - x coordinate 
15        * @param y - y coordinate 
16        */ 
17       public Point(double x, double y) { 
18           this.x = x; 
19           this.y = y; 
20       } 
21    
22       /** 
23        * return the distance of this point to the other point. 
24        * 
25        * @param other - other point 
26        * @return distance of two points 
27        */ 
28       public double distance(Point other) { 
29           return Math.sqrt((this.x - other.getX()) * (this.x - other.getX()) 
30                   + (this.y - other.getY()) * (this.y - other.getY())); 
31       } 
32    
33       /** 
34        * check if points are the same. 
35        * 
36        * @param other - other point 
37        * @return true is the points are equal, false otherwise 
38        */ 
39       public boolean equals(Point other) { 
40           return this.x == other.getX() && this.y == other.getY(); 
41       } 
42    
43       /** 
44        * return x value of the point. 
45        * 
46        * @return this.x 
47        */ 
48       public double getX() { 
49           return this.x; 
50       } 
51    
52       /** 
53        * return y value of the point. 
54        * 
55        * @return this.y 
56        */ 
57       public double getY() { 
58           return this.y; 
59       } 
60   }
