10   public class BouncingBallAnimation { 
11       /** 
12        * draw a moving ball on a draw surface. 
13        * create a new  ball and gui objects , draw the ball in different 
14        * position according to ball's velocity in 50 milliseconds gap 
15        * 
16        * @param args - strings array 
17        */ 
18       public static void main(String[] args) { 
19           GameEnvironment gameEnvironment = new GameEnvironment(); 
20           GUI gui = new GUI("ball", 200, 200); 
21           Sleeper sleeper = new Sleeper(); 
22           Ball ball = new Ball(59, 100, 30, java.awt.Color.BLACK, gameEnvironment); 
23           ball.setVelocity(2, 2); 
24           while (true) { 
25               ball.moveOneStep(1.0 / 60); 
26               DrawSurface d = gui.getDrawSurface(); 
27               ball.drawOn(d); 
28               gui.show(d); 
29               sleeper.sleepFor(50);  // wait for 50 milliseconds. 
30           } 
31       } 
32   }
