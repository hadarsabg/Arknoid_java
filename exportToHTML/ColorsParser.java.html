10       /** 
11        * parse color definition and return the specified color. 
12        * 
13        * @param s a string that represent a color 
14        * @return color 
15        */ 
16       public Color colorFromString(String s) { 
17           Color color = null; 
18           //if color is given by name of a color- 
19           try { 
20               Field field = Class.forName("java.awt.Color").getField(s); 
21               color = (Color) field.get(null); 
22           } catch (Exception e) { 
23               // color = null; 
24               //if color is given as rgb args- 
25               if (s.startsWith("RGB")) { 
26                   String rgbString = s.substring(4, s.length() - 1); 
27                   String[] rgb = rgbString.split(","); 
28                   color = new Color(Integer.parseInt(rgb[0]), 
29                           Integer.parseInt(rgb[1]), 
30                           Integer.parseInt(rgb[2])); 
31               } 
32           } 
33           return color; 
34       }
